94 const int NPE = CommonParameters::NPE();

103 std::vector<Field_G> Fmunu_1x1(6);
104
105 int i_munu = 0;
106
107 for (int mu = 0; mu < Ndim; ++mu) {
108 for (int nu = mu + 1; nu < Ndim; ++nu) {

110

113 }
