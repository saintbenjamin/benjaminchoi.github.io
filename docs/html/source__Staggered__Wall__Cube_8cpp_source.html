150 int ex = 0;

152 // eta is position in each spatial 2x2x2 cube.

167
168 if (ipet_src == ipe_t) {
169 for (int ispc = is; ispc < ns; ++ispc) {
170 int ix = ispc % Nx;
171 int iy = (ispc / Nx) % Ny;
172 int iz = ispc / (Nx * Ny);
173 int lx = ix + Nx * ipe_x;
174 int ly = iy + Ny * ipe_y;
175 int lz = iz + Nz * ipe_z;
176 int kx = lx % 2;
177 int ky = ly % 2;
178 int kz = lz % 2;
179 if ((kx == eta_x) && (ky == eta_y) && (kz == eta_z)) {
180 int site = m_index.site(ix, iy, iz, it_src);
