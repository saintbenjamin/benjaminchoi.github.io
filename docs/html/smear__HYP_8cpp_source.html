114void Smear_HYP::step1()
115{
116 for (int mu = 0; mu < m_Ndim; ++mu) {
117 for (int nu = 0; nu < m_Ndim; ++nu) {
118 if (nu == mu) continue;
119
120 for (int rho = nu + 1; rho < m_Ndim; ++rho) {
121 if (rho == mu) continue;
122
123 int sig = 6 - mu - nu - rho;
124
125 Field_G c1;
126 staple(c1, m_U[mu], m_U[sig], mu, sig);
127 //c1 *= m_alpha3 / 2.0;
128 scal(c1, m_alpha3 * 0.5);
129 m_proj->project(m_v1[index_v1(mu, nu, rho)], m_alpha3, c1, m_U[mu]);
130 }
131 }
132 }
133}
