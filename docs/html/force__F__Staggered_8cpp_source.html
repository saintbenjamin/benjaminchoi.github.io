150void Force_F_Staggered::force_udiv1(Field_G& force,
151 const Field_F_1spinor& zeta,
152 const Field_F_1spinor& eta)
153{
154 int Nc = CommonParameters::Nc();
155 int Nvol = CommonParameters::Nvol();
156 int Ndim = CommonParameters::Ndim();
157
158 Field_F_1spinor eta2(Nvol, 1);
159 Mat_SU_N ut(Nc);
160 double utr, uti;
161
162 for (int mu = 0; mu < Ndim; ++mu) {
163 m_shift.backward(eta2, eta, m_boundary[mu], mu);
164 m_fopr_ks->mult_staggered_phase(eta2, mu);
165 m_fopr_ks->mult_gm5(eta2);
166
167 for (int site = 0; site < Nvol; ++site) {
168 for (int c1 = 0; c1 < Nc; ++c1) {
169 for (int c2 = 0; c2 < Nc; ++c2) {
170 utr = zeta.cmp_r(c2, site) * eta2.cmp_r(c1, site)
171 + zeta.cmp_i(c2, site) * eta2.cmp_i(c1, site);
172 uti = zeta.cmp_r(c2, site) * eta2.cmp_i(c1, site)
173 - zeta.cmp_i(c2, site) * eta2.cmp_r(c1, site);
174 // hopping normalization
175 // utr *= 0.5/m_mq;
176 // uti *= 0.5/m_mq;
177 // mass normalization
178 utr *= 0.5;
179 uti *= 0.5;
180 ut.set(c1, c2, utr, uti);
181 }
182 }
183
184 force.set_mat(site, mu, ut);
185 }
186 }
187}
