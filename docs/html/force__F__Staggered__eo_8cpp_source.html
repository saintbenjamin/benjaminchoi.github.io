121void Force_F_Staggered_eo::force_core(Field& force_, const Field& eta_)
122{
123 const int Nc = CommonParameters::Nc();
124 const int Nvol = CommonParameters::Nvol();
125 const int Ndim = CommonParameters::Ndim();
126 const int Nvol2 = Nvol / 2;
127
128 assert(eta_.nin() == 2 * Nc);
129 assert(eta_.nvol() == Nvol2);
130 assert(eta_.nex() == 1);
131
132 Field_F_1spinor zeta(Nvol2, 1);
133 Field_F_1spinor eta(eta_);
134 //m_fopr_ks->Meo(zeta, eta, 1);
135 m_fopr_ks->mult(zeta, eta, "Doe");
136
137 Field_G force_e(Nvol2, Ndim);
138 force_udiv1(force_e, eta, zeta, 0);
139
140 Field_G force_o(Nvol2, Ndim);
141 force_udiv1(force_o, zeta, eta, 1);
142 scal(force_o, -1.0); // force_o *= -1.0;
143
144 Field_G force_eo(Nvol, Ndim);
145 for (int mu = 0; mu < Ndim; ++mu) {
146 for (int site = 0; site < Nvol2; ++site) {
147 int gx = m_index_eo.site(site, 0);
148
149 Mat_SU_N ut(Nc);
150 ut = m_Ueo->mat(gx, mu) * force_e.mat(site, mu);
151 ut.at();
152 ut *= -2.0;
153 force_eo.set_mat(gx, mu, ut);
154
155 gx = m_index_eo.site(site, 1);
156 ut = m_Ueo->mat(gx, mu) * force_o.mat(site, mu);
157 ut.at();
158 ut *= -2.0;
159 force_eo.set_mat(gx, mu, ut);
160 }
161 }
162
163 Field_G force(Nvol, Ndim);
164 m_index_eo.reverseField(force, force_eo);
165
166 copy(force_, force); // force_ = force;
167}

203void Force_F_Staggered_eo::force_udiv1(Field_G& force_h,
204 const Field_F_1spinor& zeta, const Field_F_1spinor& eta, int ieo)
205{
206 const int Nc = CommonParameters::Nc();
207 const int Nvol = CommonParameters::Nvol();
208 const int Ndim = CommonParameters::Ndim();
209 const int Nvol2 = Nvol / 2;
210
211
212
213 for (int mu = 0; mu < Ndim; ++mu) {
214 Field_F_1spinor eta2(Nvol2, 1);
215 m_shift_eo.backward_h(eta2, eta, m_boundary[mu], mu, ieo);
216 m_fopr_ks->mult_staggered_phase(eta2, mu, ieo);
217
218 for (int site = 0; site < Nvol2; ++site) {
219 Mat_SU_N ut(Nc);
220
221 for (int c1 = 0; c1 < Nc; ++c1) {
222 for (int c2 = 0; c2 < Nc; ++c2) {
223 double ut_r = zeta.cmp_r(c2, site) * eta2.cmp_r(c1, site)
224 + zeta.cmp_i(c2, site) * eta2.cmp_i(c1, site);
225 double ut_i = zeta.cmp_r(c2, site) * eta2.cmp_i(c1, site)
226 - zeta.cmp_i(c2, site) * eta2.cmp_r(c1, site);
227 // ut_r *= -0.5 / m_mq; // hopping normalization
228 // ut_i *= -0.5 / m_mq;
229 ut_r *= -0.5; // mass normalization
230 ut_i *= -0.5;
231
232 ut.set(c1, c2, ut_r, ut_i);
233 }
234 }
235
236 force_h.set_mat(site, mu, ut);
237 }
238 }
239}
