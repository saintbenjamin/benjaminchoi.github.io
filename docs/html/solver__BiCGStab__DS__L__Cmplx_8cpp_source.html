228void Solver_BiCGStab_DS_L_Cmplx::solve(Field& xq, const Field& b,
229 int& Nconv, double& diff)
230{
231 const double bnorm2 = b.norm2();
232 const int bsize = b.size();
233
234 vout.paranoiac(m_vl, "%s: starts\n", class_name.c_str());
235 vout.paranoiac(m_vl, " norm of b = %16.8e\n", bnorm2);
236 vout.paranoiac(m_vl, " size of b = %d\n", bsize);
237
238 bool is_converged = false;
239 int Nconv2 = 0;
240 double diff2 = 1.0; // superficial initialization
241 double rr;
242
243 int Nconv_unit = 1;
244 // if (m_fopr->get_mode() == "DdagD" || m_fopr->get_mode() == "DDdag") {
245 // Nconv_unit = 2;
246 // }
247
248 reset_field(b);
249
250 if (m_use_init_guess) {
251 copy(m_s, xq); // s = xq;
252 } else {
253 copy(m_s, b); // s = b;
254 }
255 solve_init(b, rr);
256 Nconv2 += Nconv_unit;
257
258 vout.detailed(m_vl, " iter: %8d %22.15e\n", Nconv2, rr / bnorm2);
259
260
261 for (int i_restart = 0; i_restart < m_Nrestart; i_restart++) {
262 for (int iter = 0; iter < m_Niter; iter++) {
263 if (rr / bnorm2 < m_Stop_cond) break;
264
265 solve_step(rr);
266 Nconv2 += 2 * Nconv_unit * m_N_L_prev;
267
268 vout.paranoiac(m_vl, " iter,N_L: %8d %8d\n", Nconv2, m_N_L_prev);
269 vout.detailed(m_vl, " iter: %8d %22.15e\n", Nconv2, rr / bnorm2);
270 }
271
272 //- calculate true residual
273 m_fopr->mult(m_s, m_x); // s = m_fopr->mult(x);
274 axpy(m_s, -1.0, b); // s -= b;
275 diff2 = m_s.norm2();
276
277 if (diff2 / bnorm2 < m_Stop_cond) {
278 vout.detailed(m_vl, "%s: converged.\n", class_name.c_str());
279 vout.detailed(m_vl, " iter(final): %8d %22.15e\n", Nconv2, diff2 / bnorm2);
280
281 is_converged = true;
282
283 m_Nrestart_count = i_restart;
284 m_Nconv_count = Nconv2;
285
286 break;
287 } else {
288 //- restart with new approximate solution
289 copy(m_s, m_x); // s = x;
290 solve_init(b, rr);
291
292 vout.detailed(m_vl, "%s: restarted.\n", class_name.c_str());
293 }
294 }
295
296
297 if (!is_converged) {
298 vout.crucial(m_vl, "Error at %s: not converged.\n", class_name.c_str());
299 vout.crucial(m_vl, " iter(final): %8d %22.15e\n", Nconv2, diff2 / bnorm2);
300 exit(EXIT_FAILURE);
301 }
302
303
304 copy(xq, m_x); // xq = x;
305
306#pragma omp barrier
307#pragma omp master
308 {
309 diff = sqrt(diff2 / bnorm2);
310 Nconv = Nconv2;
311 }
312#pragma omp barrier
313}

380void Solver_BiCGStab_DS_L_Cmplx::solve_step(double& rr)
381{
382 dcomplex rho_prev2 = m_rho_prev;
383 dcomplex alpha_prev2 = m_alpha_prev;
384
385 int N_L_tmp = 0; // superficial initialization
386 dcomplex c_Rayleigh_prev = cmplx(0.0, 0.0);
387
388 bool is_converged_L = false;
389
390
391 for (int j = 0; j < m_N_L_prev; ++j) {
392 if (!is_converged_L) {
393 dcomplex rho = dotc(m_r[j], m_r_init); // rho = r[j] * r_init;
394 rho = conj(rho);
395
396 dcomplex beta = alpha_prev2 * (rho / rho_prev2);
397
398 rho_prev2 = rho;
399
400 for (int i = 0; i < j + 1; ++i) {
401 aypx(-beta, m_u[i], m_r[i]); // u[i] = - beta * u[i] + r[i];
402 }
403
404 m_fopr->mult(m_u[j + 1], m_u[j]); // u[j+1] = m_fopr->mult(u[j]);
405
406 dcomplex gamma = dotc(m_u[j + 1], m_r_init);
407 alpha_prev2 = rho_prev2 / conj(gamma);
408
409 for (int i = 0; i < j + 1; ++i) {
410 axpy(m_r[i], -alpha_prev2, m_u[i + 1]); // r[i] -= alpha_prev2 * u[i+1];
411 }
412
413 m_fopr->mult(m_r[j + 1], m_r[j]); // r[j+1] = m_fopr->mult(r[j]);
414
415 axpy(m_x, alpha_prev2, m_u[0]); // x += alpha_prev2 * u[0];
416
417
418 //- calculate the Rayleigh quotient for N_L_tmp
419
420 // dcomplex c_Rayleigh = (r[j] * r[j+1]) / (r[j] * r[j]);
421 double r_tmp = m_r[j].norm2(); // r_tmp = r[j] * r[j];
422 dcomplex c_Rayleigh = dotc(m_r[j], m_r[j + 1]) / r_tmp; // c_Rayleigh = r[j] * r[j+1] / r_tmp;
423
424 dcomplex c_E = (c_Rayleigh - c_Rayleigh_prev) / c_Rayleigh;
425
426 c_Rayleigh_prev = c_Rayleigh;
427
428 N_L_tmp = j + 1;
429
430 // vout.paranoiac(m_vl, "N_L_tmp,abs(c_E),m_Tol_L = %d %f %f\n",N_L_tmp,abs(c_E),m_Tol_L);
431
432 if (abs(c_E) < m_Tol_L) {
433 // vout.paranoiac(m_vl, "N_L_tmp = %d\n",N_L_tmp);
434
435 // break;
436 is_converged_L = true;
437 }
438 }
439 }
440
441
442 std::vector<double> sigma(m_N_L + 1);
443 std::vector<dcomplex> gamma_prime(m_N_L + 1);
444
445 // NB. tau(m_N_L,m_N_L+1), not (m_N_L+1,m_N_L+1)
446 std::vector<dcomplex> tau(m_N_L * (m_N_L + 1));
447
448 const double sigma_0 = m_r[0].norm2();
449
450 for (int j = 1; j < N_L_tmp + 1; ++j) {
451 for (int i = 1; i < j; ++i) {
452 int ij = index_ij(i, j);
453
454 dcomplex r_ji = dotc(m_r[j], m_r[i]);
455 tau[ij] = conj(r_ji) / sigma[i]; // tau[ij] = (r[j] * r[i]) / sigma[i];
456 axpy(m_r[j], -tau[ij], m_r[i]); // r[j] -= tau[ij] * r[i];
457 }
458
459 sigma[j] = m_r[j].norm2(); // sigma[j] = r[j] * r[j];
460
461 dcomplex r_0j = dotc(m_r[0], m_r[j]);
462 gamma_prime[j] = conj(r_0j) / sigma[j]; // gamma_prime[j] = (r[0] * r[j]) / sigma[j];
463
464 //- a prescription to improve stability of BiCGStab(L)
465 double abs_rho = abs(r_0j) / sqrt(sigma[j] * sigma_0);
466 if (abs_rho < m_Omega_tolerance) {
467 gamma_prime[j] *= m_Omega_tolerance / abs_rho;
468 }
469 }
470
471
472 std::vector<dcomplex> gamma(m_N_L + 1);
473 gamma[N_L_tmp] = gamma_prime[N_L_tmp];
474
475 for (int j = N_L_tmp - 1; j > 0; --j) {
476 dcomplex c_tmp = cmplx(0.0, 0.0);
477
478 for (int i = j + 1; i < N_L_tmp + 1; ++i) {
479 int ji = index_ij(j, i);
480 c_tmp += tau[ji] * gamma[i];
481 }
482
483 gamma[j] = gamma_prime[j] - c_tmp;
484 }
485
486
487 // NB. gamma_double_prime(m_N_L), not (m_N_L+1)
488 std::vector<dcomplex> gamma_double_prime(m_N_L);
489
490 for (int j = 1; j < N_L_tmp; ++j) {
491 dcomplex c_tmp = cmplx(0.0, 0.0);
492
493 for (int i = j + 1; i < N_L_tmp; ++i) {
494 int ji = index_ij(j, i);
495 c_tmp += tau[ji] * gamma[i + 1];
496 }
497
498 gamma_double_prime[j] = gamma[j + 1] + c_tmp;
499 }
500
501
502 axpy(m_x, gamma[1], m_r[0]); // x += gamma[ 1] * r[ 0];
503 axpy(m_r[0], -gamma_prime[N_L_tmp], m_r[N_L_tmp]); // r[0] -= gamma_prime[N_L_tmp] * r[N_L_tmp];
504 axpy(m_u[0], -gamma[N_L_tmp], m_u[N_L_tmp]); // u[0] -= gamma[ N_L_tmp] * u[N_L_tmp];
505
506 for (int j = 1; j < N_L_tmp; ++j) {
507 axpy(m_x, gamma_double_prime[j], m_r[j]); // x += gamma_double_prime[j] * r[j];
508 axpy(m_r[0], -gamma_prime[j], m_r[j]); // r[0] -= gamma_prime[ j] * r[j];
509 axpy(m_u[0], -gamma[j], m_u[j]); // u[0] -= gamma[ j] * u[j];
510 }
511
512 rr = m_r[0].norm2(); // rr = r[0] * r[0];
513
514
515#pragma omp barrier
516#pragma omp master
517 {
518 m_rho_prev = rho_prev2;
519 m_alpha_prev = alpha_prev2;
520 m_N_L_prev = N_L_tmp;
521
522 m_rho_prev *= -gamma_prime[N_L_tmp];
523 }
524#pragma omp barrier
525}

529double Solver_BiCGStab_DS_L_Cmplx::flop_count()
530{
531 const int NPE = CommonParameters::NPE();
532
533 //- NB1 Nin = 2 * Nc * Nd, Nex = 1 for field_F
534 //- NB2 Nvol = CommonParameters::Nvol()/2 for eo
535 const int Nin = m_x.nin();
536 const int Nvol = m_x.nvol();
537 const int Nex = m_x.nex();
538
539 const double gflop_fopr = m_fopr->flop_count();
540
541 if (gflop_fopr < CommonParameters::epsilon_criterion()) {
542 vout.crucial(m_vl, "Warning at %s: no fopr->flop_count() is available, setting flop = 0\n", class_name.c_str());
543 return 0.0;
544 }
545
546 const double gflop_axpy = (Nin * Nex * 2) * ((Nvol * NPE) / 1.0e+9);
547 const double gflop_dotc = (Nin * Nex * 4) * ((Nvol * NPE) / 1.0e+9);
548 const double gflop_norm = (Nin * Nex * 2) * ((Nvol * NPE) / 1.0e+9);
549
550 const int N_L_prev_total = (m_Nconv_count - 1) / 2;
551
552 const double gflop_init = gflop_fopr + gflop_axpy + gflop_norm;
553 const double gflop_step_BiCG_part = 2 * N_L_prev_total * gflop_fopr
554 + 3 * N_L_prev_total * gflop_dotc
555 + (N_L_prev_total + 2 * m_N_L_part_count) * gflop_axpy
556 + N_L_prev_total * gflop_norm;
557 const double gflop_step_L_part = (m_N_L_part_count + N_L_prev_total) * gflop_dotc
558 + (m_N_L_part_count + 3 * N_L_prev_total) * gflop_axpy
559 + (N_L_prev_total + m_Niter_count) * gflop_norm;
560 const double gflop_step = gflop_step_BiCG_part + gflop_step_L_part;
561 const double gflop_true_residual = gflop_fopr + gflop_axpy + gflop_norm;
562
563 const double gflop = gflop_norm + gflop_init + gflop_step + gflop_true_residual * (m_Nrestart_count + 1)
564 + gflop_init * m_Nrestart_count;
565
566
567 return gflop;
568}
